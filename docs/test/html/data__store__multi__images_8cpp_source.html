80 void data_store_multi_images::get_file_sizes() {
81  std::vector<Triple> my_file_sizes(m_my_datastore_indices.size()*m_num_img_srcs);
82  if (m_master) std::cerr << "STARTING " << get_name() << "::get_file_sizes for " << my_file_sizes.size() << " files\n";
83 
84  size_t cur_offset = 0;
85  std::unordered_map<std::string, size_t> names;
86  size_t jj = 0;
87  for (auto base_index : m_my_datastore_indices) {
88  const std::vector<std::string> sample(get_sample(base_index));
89  for (size_t k=0; k<sample.size(); k++) {
90  size_t index = base_index*m_num_img_srcs + k;
91  size_t file_len = 0;
92  if (names.find(sample[k]) != names.end()) {
93  file_len = names[sample[k]];
94  } else {
95  file_len = get_file_size(m_dir, sample[k]);
96  names[sample[k]] = file_len;
97  }
98 
99  my_file_sizes[jj].global_index = index;
100  my_file_sizes[jj].num_bytes = file_len;
101  my_file_sizes[jj].offset = cur_offset;
102  my_file_sizes[jj].rank = m_rank;
103  cur_offset += my_file_sizes[jj].num_bytes;
104 
105  if (my_file_sizes[jj].num_bytes == 0) {
106  std::stringstream err;
107  err << __FILE__ << " " << __LINE__ << " :: "
108  << " file size is 0";
109  throw lbann_exception(err.str());
110  }
111  ++jj;
112  }
113  }
114 
115  exchange_file_sizes(my_file_sizes, m_num_global_indices*m_num_img_srcs);
116 }

118 void data_store_multi_images::read_files() {
119  std::stringstream err;
120  for (auto base_index : m_my_datastore_indices) {
121  const std::vector<std::string> sample(get_sample(base_index));
122  for (size_t k=0; k<sample.size(); k++) {
123  size_t index = base_index * m_num_img_srcs + k;
124 
125  if (m_offsets.find(index) == m_offsets.end()) {
126  err << __FILE__ << " " << __LINE__ << " :: "
127  << " m_offsets.find(index) failed for index: " << index;
128  throw lbann_exception(err.str());
129  }
130  size_t offset = m_offsets[index];
131 
132  if (m_file_sizes.find(index) == m_file_sizes.end()) {
133  err << __FILE__ << " " << __LINE__ << " :: "
134  << " m_file_sizes.find(index) failed for index: " << index;
135  throw lbann_exception(err.str());
136  }
137  size_t file_len = m_file_sizes[index];
138 
139  if (offset + file_len > m_data.size()) {
140  err << __FILE__ << " " << __LINE__ << " :: "
141  << " of " << m_my_datastore_indices.size() << " offset: " << offset
142  << " file_len: " << file_len << " offset+file_len: "
143  << offset+file_len << " m_data.size(): " << m_data.size()
144  << "\noffset+file_len must be <= m_data.size()";
145  throw lbann_exception(err.str());
146  }
147 
148  load_file(m_dir, sample[k], m_data.data()+offset, file_len);
149  }
150  }
151 }

154 void data_store_multi_images::extended_testing() {
155  if (m_master) std::cerr << "STARTING data_store_multi_images::extended_testing()\n";
156  std::stringstream err;
157  std::vector<unsigned char> v;
158  for (auto idx : m_my_minibatch_indices_v) {
159  int base_index = (*m_shuffled_indices)[idx];
160  const std::vector<std::string> sample(get_sample(base_index));
161  for (size_t k=0; k<sample.size(); k++) {
162  size_t index = base_index*m_num_img_srcs + k;
163 
164  if (m_file_sizes.find(index) == m_file_sizes.end()) {
165  err << __FILE__ << " " << __LINE__ << " :: "
166  << " file length not found: " << index;
167  throw lbann_exception(err.str());
168  }
169  size_t file_len = m_file_sizes[index];
170 
171  v.resize(file_len);
172  load_file(m_dir, sample[k], v.data(), file_len);
173 
174  if (m_my_minibatch_data.find(index) == m_my_minibatch_data.end()) {
175  err << __FILE__ << " " << __LINE__ << " :: "
176  << " m_my_minibatch_data.find(" << index << ") failed.";
177  throw lbann_exception(err.str());
178  }
179  if (m_my_minibatch_data[index] != v) {
180  err << __FILE__ << " " << __LINE__ << " :: "
181  << " data_store_multi_images::extended_testing: "
182  << " rank: " << m_rank << " index: " << index << " FAILED!\n";
183  throw lbann_exception(err.str());
184  }
185  }
186  }
187  std::cerr << "rank: " << m_rank << " data_store_multi_images::extended_testing, PASSED!\n";
188 }
