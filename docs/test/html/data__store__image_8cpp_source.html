108 void data_store_image::load_file(const std::string &dir, const std::string &fn, unsigned char *p, size_t sz) {
109  std::string imagepath;
110  if (dir != "") {
111  imagepath = dir + fn;
112  } else {
113  imagepath = fn;
114  }
115  std::ifstream in(imagepath.c_str(), std::ios::in | std::ios::binary);
116  if (!in) {
117  std::stringstream err;
118  err << __FILE__ << " " << __LINE__ << " :: "
119  << "failed to open " << imagepath << " for reading"
120  << "; dir: " << dir << " fn: " << fn;
121  throw lbann_exception(err.str());
122  }
123  in.read((char*)p, sz);
124  if ((int)sz != in.gcount()) {
125  std::stringstream err;
126  err << __FILE__ << " " << __LINE__ << " :: "
127  << "failed to read " << sz << " bytes from " << imagepath
128  << " num bytes read: " << in.gcount();
129  throw lbann_exception(err.str());
130  }
131  in.close();
132 }

134 void data_store_image::exchange_data() {
135  if (m_master) std::cerr << "starting exchange_data\n";
136  std::stringstream err;
137 
138  //build map: proc -> global indices that proc needs for this epoch, and
139  // which I own
140  std::unordered_map<int, std::unordered_set<int>> proc_to_indices;
141  for (size_t p=0; p<m_all_minibatch_indices.size(); p++) {
142  for (auto idx : m_all_minibatch_indices[p]) {
143  int index = (*m_shuffled_indices)[idx];
144  if (m_my_datastore_indices.find(index) != m_my_datastore_indices.end()) {
145  proc_to_indices[p].insert(index);
146  }
147  }
148  }
149 
150  //start sends
151  std::vector<std::vector<MPI_Request>> send_req(m_np);
152  std::vector<std::vector<MPI_Status>> send_status(m_np);
153  for (int p=0; p<m_np; p++) {
154  send_req[p].resize(proc_to_indices[p].size()*m_num_img_srcs);
155  send_status[p].resize(proc_to_indices[p].size()*m_num_img_srcs);
156  size_t jj = 0;
157  for (auto idx : proc_to_indices[p]) {
158  for (size_t k=0; k<m_num_img_srcs; k++) {
159  int index = idx*m_num_img_srcs+k;
160  if (m_file_sizes.find(index) == m_file_sizes.end()) {
161  err << __FILE__ << " " << __LINE__ << " :: "
162  << " m_file_sizes.find(" << index << ") failed";
163  throw lbann_exception(err.str());
164  }
165  if (m_offsets.find(index) == m_offsets.end()) {
166  err << __FILE__ << " " << __LINE__ << " :: "
167  << " m_offets.find(" << index << ") failed";
168  throw lbann_exception(err.str());
169  }
170  int len = m_file_sizes[index];
171  size_t offset = m_offsets[index];
172  MPI_Isend(m_data.data()+offset, len, MPI_BYTE, p, index, m_mpi_comm, &(send_req[p][jj++]));
173  }
174  }
175  if (jj != send_req[p].size()) throw lbann_exception("ERROR 1");
176  } //start sends
177 
178 
179  //build map: proc -> global indices that proc owns that I need
180  proc_to_indices.clear();
181  //note: datastore indices are global; no need to consult shuffled_indices
182  for (auto idx : m_my_minibatch_indices_v) {
183  int index = (*m_shuffled_indices)[idx];
184  int owner = get_index_owner(index);
185  proc_to_indices[owner].insert(index);
186  }
187 
188  //start recvs
189  m_my_minibatch_data.clear();
190  std::vector<std::vector<MPI_Request>> recv_req(m_np);
191  std::vector<std::vector<MPI_Status>> recv_status(m_np);
192  for (auto t : proc_to_indices) {
193  int owner = t.first;
194  size_t jj = 0;
195  const std::unordered_set<int> &s = t.second;
196  recv_req[owner].resize(s.size()*m_num_img_srcs);
197  recv_status[owner].resize(s.size()*m_num_img_srcs);
198  for (auto idx : s) {
199  for (size_t k=0; k<m_num_img_srcs; k++) {
200  size_t index = idx*m_num_img_srcs+k;
201  if (m_file_sizes.find(index) == m_file_sizes.end()) {
202  err << __FILE__ << " " << __LINE__ << " :: "
203  << " m_file_sizes.find(" << index << ") failed"
204  << " m_file_sizes.size(): " << m_file_sizes.size()
205  << " m_my_minibatch_indices_v.size(): " << m_my_minibatch_indices_v.size();
206  }
207  size_t len = m_file_sizes[index];
208  m_my_minibatch_data[index].resize(len);
209  MPI_Irecv(m_my_minibatch_data[index].data(), len, MPI_BYTE, owner, index, m_mpi_comm, &(recv_req[owner][jj++]));
210  }
211  }
212  }
213 
214  //wait for sends to finish
215  for (size_t i=0; i<send_req.size(); i++) {
216  MPI_Waitall(send_req[i].size(), send_req[i].data(), send_status[i].data());
217  }
218 
219  //wait for recvs to finish
220  for (size_t i=0; i<recv_req.size(); i++) {
221  MPI_Waitall(recv_req[i].size(), recv_req[i].data(), recv_status[i].data());
222  }
223 }

226 void data_store_image::exchange_file_sizes(std::vector<Triple> &my_file_sizes, int num_global_indices) {
227  std::vector<int> num_bytes(m_np);
228  int bytes = my_file_sizes.size()*sizeof(Triple);
229  MPI_Allgather(&bytes, 1, MPI_INT,
230  num_bytes.data(), 1, MPI_INT,
231  m_mpi_comm);
232 
233  std::vector<int> disp(m_num_readers);
234  disp[0] = 0;
235  for (int h=1; h<(int)m_num_readers; h++) {
236  disp[h] = disp[h-1] + num_bytes[h-1];
237  }
238  std::vector<Triple> global_file_sizes(num_global_indices);
239 
240  //@todo: couldn't get m_comm->model_gatherv to work
241  //m_comm->model_gatherv(&my_file_sizes[0], my_file_sizes.size(),
242  // &global_file_sizes[0], &num_images[0], &disp[0]);
243  MPI_Allgatherv(my_file_sizes.data(), my_file_sizes.size()*sizeof(Triple), MPI_BYTE,
244  global_file_sizes.data(), num_bytes.data(), disp.data(), MPI_BYTE,
245  m_mpi_comm);
246 
247  size_t j = 0;
248  for (auto t : global_file_sizes) {
249  m_file_sizes[t.global_index] = t.num_bytes;
250  m_offsets[t.global_index] = t.offset;
251  if (t.num_bytes <= 0) {
252  std::stringstream err;
253  err << __FILE__ << " " << __LINE__ << " :: "
254  << "num_bytes <= 0 (" << t.num_bytes << ")"
255  << " for # " << j << " of " << global_file_sizes.size();
256  throw lbann_exception(err.str());
257  }
258  //m_owner_mapping[t.global_index] = t.rank;
259  ++j;
260  }
261 }
