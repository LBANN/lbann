93 void generic_data_store::setup() {
94  set_shuffled_indices( &(m_reader->get_shuffled_indices()) );
95  set_num_global_indices();
96  m_num_readers = m_reader->get_num_parallel_readers();
97  if (m_master) {
98  std::cerr << "data_reader type is: " << m_reader->get_type() << "\n";
99  }
100 
101  // get the set of global indices used by this processor in
102  // generic_data_reader::fetch_data(). Note that these are
103  // "original' indices, not shuffled indices, i.e, these indices
104  // remain constant through all epochs
105  if (m_collect_minibatch_indices) {
106  if (m_master) { std::cerr << "calling m_model->collect_indices\n"; }
107  m_reader->set_save_minibatch_entries(true);
108  if (m_reader->get_role() == "train") {
109  m_model->collect_indices(execution_mode::training);
110  } else if (m_reader->get_role() == "validate") {
111  m_model->collect_indices(execution_mode::validation);
112  } else if (m_reader->get_role() == "test") {
113  m_model->collect_indices(execution_mode::testing);
114  } else {
115  std::stringstream s2;
116  s2 << __FILE__ << " " << __LINE__ << " :: "
117  << " bad role; should be train, test, or validate;"
118  << " we got: " << m_reader->get_role();
119  throw lbann_exception(s2.str());
120  }
121  m_reader->set_save_minibatch_entries(false);
122  }
123  m_my_minibatch_indices = &(m_reader->get_minibatch_indices());
124 }

173 void generic_data_store::exchange_mb_indices() {
174  exchange_mb_counts();
175  //setup data structures to exchange minibatch indices with all processors
176  //displacement vector
177  std::vector<int> displ(m_np);
178  displ[0] = 0;
179  for (size_t j=1; j<m_mb_counts.size(); j++) {
180  displ[j] = displ[j-1] + m_mb_counts[j-1];
181  }
182 
183  //recv vector
184  int n = std::accumulate(m_mb_counts.begin(), m_mb_counts.end(), 0);
185  std::vector<int> all_indices(n);
186 
187  //receive the indices
188  MPI_Allgatherv(
189  m_my_minibatch_indices_v.data(), m_my_minibatch_indices_v.size(), MPI_INT,
190  all_indices.data(), m_mb_counts.data(), displ.data(),
191  MPI_INT, m_mpi_comm);
192 
193  //fill in the final data structure
194  m_all_minibatch_indices.resize(m_np);
195  for (int j=0; j<m_np; j++) {
196  m_all_minibatch_indices[j].reserve(m_mb_counts[j]);
197  for (int i=displ[j]; i<displ[j]+m_mb_counts[j]; i++) {
198  m_all_minibatch_indices[j].push_back(all_indices[i]);
199  }
200  }
201 }
