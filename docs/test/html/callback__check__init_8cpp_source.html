34 void lbann_callback_check_init::on_train_begin(model *m) {
35  // Skip after the first epoch.
36  if (m->get_cur_epoch() != 0) {
37  return;
38  }
39  lbann_comm *comm = m->get_comm();
40  if (comm->am_world_master()) {
41  std::cout << "Checking all model initial weights match..." << std::endl;
42  }
43  if (comm->get_num_models() == 1) {
44  return;
45  }
46 
47  for (const auto w : m->get_weights()) {
48  if (comm->am_world_master()) {
49  std::cout << "Checking " << w->get_name() << std::endl;
50  }
51  // Model 0 holds the master copy, it gathers the values from other models
52  // and compares them.
53  const Mat& local_matrix = w->get_values().LockedMatrix();
54  Mat remote_matrix(local_matrix.Height(), local_matrix.Width());
55  for (int model = 1; model < comm->get_num_models(); ++model) {
56  comm->global_barrier();
57  if (comm->get_model_rank() == 0) {
58  comm->recv(remote_matrix, model);
59  if (!check_equal(local_matrix, remote_matrix)) {
60  std::stringstream ss;
61  ss << "check_init: "
62  << "model " << model << " "
63  << "rank in model " << comm->get_rank_in_model() << " "
64  << "does not match model 0";
65  throw lbann_exception(ss.str());
66  }
67  } else if (comm->get_model_rank() == model) {
68  comm->send(local_matrix, 0);
69  }
70  }
71  }
72 }

74 bool lbann_callback_check_init::check_equal(const Mat& x, const Mat& y) const {
75  const El::Int height = x.Height();
76  const El::Int width = x.Width();
77  if (height != y.Height() || width != y.Width() || x.LDim() != y.LDim()) {
78  return false;
79  }
80  const DataType *x_buf = x.LockedBuffer();
81  const DataType *y_buf = y.LockedBuffer();
82  for (El::Int i = 0; i < height * width; ++i) {
83  if (x_buf[i] != y_buf[i]) {
84  return false;
85  }
86  }
87  return true;
88 }
