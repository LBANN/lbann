32 EvalType r2_metric::evaluate_compute(const AbsDistMat& prediction,
33  const AbsDistMat& ground_truth) {
34 
35  // Get matrix dimensions
36  const int local_height = prediction.LocalHeight();
37  const int local_width = prediction.LocalWidth();
38  const int width = prediction.Width();
39 
40  // Get local matrices
41  const Mat& prediction_local = prediction.LockedMatrix();
42  const Mat& ground_truth_local = ground_truth.LockedMatrix();
43 
44  DataType gt_mean, gt_std;
45  // Entry-wise mean of ground truth
46  //@todo fix stat class not to compute stdev if not needed
47  entrywise_mean_and_stdev(ground_truth, gt_mean, gt_std);
48 
49  // Compute residual sum of squares ss_res
50  // and sum of squares ss_tot as sum(square(ground_truth - mean(ground_truth)))
51  EvalType ss_res = 0;
52  EvalType ss_tot = 0;
53  #pragma omp parallel for reduction(+:ss_res,ss_tot) collapse(2)
54  for(El::Int col = 0; col < local_width; ++col) {
55  for(El::Int row = 0; row < local_height; ++row) {
56  const EvalType true_val = ground_truth_local(row, col);
57  const EvalType pred_val = prediction_local(row, col);
58  const EvalType val1 = true_val - pred_val;
59  const EvalType val2 = true_val - gt_mean;
60  ss_res += val1 * val1;
61  ss_tot += val2 * val2;
62  }
63  }
64 
65  EvalType res_tot[2] = {ss_res, ss_tot}; // Pack to do one allreduce.
66  El::mpi::AllReduce(res_tot, 2, prediction.DistComm());
67  //Keras and TF add epsilon (1e-07) to denominator to avoid inf score
68  //We might actually need to do this here and other places too
69  EvalType ss_tot_eps = res_tot[1] + 0.0000001;
70  //Multiply by width because base class divide by mini-batch size
71  return ((1-(res_tot[0]/ss_tot_eps))*width);
72 }
