54 void lbann::lbann_callback_debug_io::print_fp_start(model *m, generic_input_layer *input) {
55  int64_t step;
56  switch(m->get_execution_mode()) {
57  case execution_mode::training:
58  step = m->get_cur_step();
59  break;
60  case execution_mode::validation:
61  step = m->get_cur_validation_step();
62  break;
63  case execution_mode::testing:
64  step = m->get_cur_testing_step();
65  break;
66  default:
67  throw lbann_exception("Illegal execution mode in evaluate forward prop function");
68  }
69  std::cout << "[" << m->get_comm()->get_model_rank()
70  << "." << m->get_comm()->get_rank_in_model()
71  << "] @" << m->get_cur_epoch() << "." << step
72  << " Phase: " << _to_string(m->get_execution_mode())
73  << " starting forward propagation for layer " << input->get_name()
74  << " type: " << input->get_type()
75  << " iteration: " << input->get_data_reader()->get_current_mini_batch_index()
76  << " of " << input->get_num_iterations_per_epoch()
77  << " loading idx " << input->get_data_reader()->get_loaded_mini_batch_index()
78  << " bs=" << input->get_current_mini_batch_size() << "/"
79  << input->get_current_global_mini_batch_size()
80  << " @" << input->get_data_reader()->get_position()
81  // << " %" << input->get_data_reader()->get_batch_stride()
82  << " ^" << input->get_data_reader()->get_sample_stride()
83  << std::endl;
84 }

86 // 179i @ 300s (=5m*60s) + 1i @ 100s (=5m*45s):offset <- num models
87 void lbann::lbann_callback_debug_io::print_phase_start(model *m, execution_mode mode) {
88 
89  // Get data reader from first input layer in model
90  generic_data_reader* data_reader = nullptr;
91  for (auto&& l : m->get_layers()) {
92  auto&& input = dynamic_cast<generic_input_layer*>(l);
93  if (input != nullptr) {
94  data_reader = input->get_data_reader(mode);
95  break;
96  }
97  }
98  if (data_reader == nullptr) { return; }
99 
100  int64_t step;
101  switch(mode) {
102  case execution_mode::training:
103  step = m->get_cur_step(); break;
104  case execution_mode::validation:
105  step = m->get_cur_validation_step(); break;
106  case execution_mode::testing:
107  step = m->get_cur_testing_step(); break;
108  default:
109  throw lbann_exception("Illegal execution mode in evaluate forward prop function");
110  }
111 
112  if(data_reader->get_rank() < data_reader->get_num_parallel_readers()) {
113  std::cout << "[" << m->get_comm()->get_model_rank()
114  << "." << m->get_comm()->get_rank_in_model()
115  << "] @" << 0 << "." << step
116  << " Starting Phase: " << _to_string(mode)
117  << " " << (data_reader->get_num_iterations_per_epoch() - 1)
118  << "i @ " << data_reader->get_global_mini_batch_size()
119  << "s (=" << m->get_comm()->get_num_models()
120  << "m *" << data_reader->get_mini_batch_size()
121  << "s [+" << data_reader->get_stride_to_next_mini_batch()
122  << "s]) + 1i @ " << data_reader->get_global_last_mini_batch_size()
123  << "s (=" << m->get_comm()->get_num_models()
124  << "m *" << data_reader->get_last_mini_batch_size()
125  << "s [+" << data_reader->get_stride_to_last_mini_batch()
126  << "s]):"
127  <<" base offset "<< data_reader->get_base_offset()
128  << " model offset " << data_reader->get_model_offset()
129  << " par. readers = " << data_reader->get_num_parallel_readers()
130  << "r"
131  << std::endl;
132  }else {
133  std::cout << "[" << m->get_comm()->get_model_rank()
134  << "." << m->get_comm()->get_rank_in_model()
135  << "] @" << 0 << "." << step
136  << " Starting Phase: " << _to_string(mode)
137  << " " << (data_reader->get_num_iterations_per_epoch())
138  << "i "
139  << " par. readers = " << data_reader->get_num_parallel_readers()
140  << "r (Inactive Reader)"
141  << std::endl;
142  }
143 }
