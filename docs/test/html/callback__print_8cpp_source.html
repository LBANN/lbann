130 void lbann_callback_print::report_results(model *m) {
131  lbann_comm *comm = m->get_comm();
132 
133  // Get string for execution mode
134  const execution_mode mode = m->get_execution_mode();
135  std::string mode_string;
136  switch (mode) {
137  case execution_mode::training:
138  mode_string = "training epoch " + std::to_string(m->get_cur_epoch()-1);
139  break;
140  case execution_mode::validation:
141  mode_string = "validation";
142  break;
143  case execution_mode::testing:
144  mode_string = "test";
145  break;
146  default:
147  std::stringstream err;
148  err << __FILE__ << " " << __LINE__ << " :: "
149  << "invalid execution mode for reporting results";
150  throw lbann_exception(err.str());
151  }
152 
153  if (comm->am_model_master()) {
154  const int num_models = comm->get_num_models();
155 
156  // Report objective function value
157  const EvalType obj_fn = m->get_objective_function()->get_mean_value(mode);
158  const int obj_fn_samples = m->get_objective_function()->get_statistics_num_samples(mode);
159  if (comm->am_world_master()) {
160  std::vector<EvalType> obj_fn_list(comm->get_num_models());
161  std::vector<int> num_samples_list(comm->get_num_models());
162  comm->intermodel_gather(obj_fn, obj_fn_list);
163  comm->intermodel_gather(obj_fn_samples, num_samples_list);
164  for (int i = 0; i < num_models; ++i) {
165  std::cout << "Model " << i << " " << mode_string << " "
166  << "objective function : " << obj_fn_list[i]
167  << std::endl;
168  }
169  if (num_models > 1) {
170  const EvalType avg_obj_fn = (std::inner_product(num_samples_list.begin(),
171  num_samples_list.end(),
172  obj_fn_list.begin(),
173  EvalType(0))
174  / std::accumulate(num_samples_list.begin(),
175  num_samples_list.end(),
176  0));
177  std::cout << "World average " << mode_string << " "
178  << "objective function : " << avg_obj_fn
179  << std::endl;
180  }
181  } else {
182  comm->intermodel_gather(obj_fn, comm->get_world_master());
183  comm->intermodel_gather(obj_fn_samples, comm->get_world_master());
184  }
185 
186  // Report score for each metric
187  for (const auto& met : m->get_metrics()) {
188  const EvalType score = met->get_mean_value(mode);
189  const int score_samples = met->get_statistics_num_samples(mode);
190  if (comm->am_world_master()) {
191  std::vector<EvalType> score_list(comm->get_num_models());
192  std::vector<int> num_samples_list(comm->get_num_models());
193  comm->intermodel_gather(score, score_list);
194  comm->intermodel_gather(score_samples, num_samples_list);
195  for (int i = 0; i < num_models; ++i) {
196  std::cout << "Model " << i << " " << mode_string << " "
197  << met->name() << " : "
198  << score_list[i] << met->get_unit()
199  << std::endl;
200  }
201  if (num_models > 1) {
202  const EvalType avg_score = (std::inner_product(num_samples_list.begin(),
203  num_samples_list.end(),
204  score_list.begin(),
205  EvalType(0))
206  / std::accumulate(num_samples_list.begin(),
207  num_samples_list.end(),
208  0));
209  std::cout << "World " << mode_string << " "
210  << met->name() << " : "
211  << avg_score << met->get_unit()
212  << std::endl;
213  }
214  } else {
215  comm->intermodel_gather(score, comm->get_intermodel_master());
216  comm->intermodel_gather(score_samples, comm->get_intermodel_master());
217  }
218  }
219 
220  }
