62 void data_reader_merge_features::load() {
63  // Load each data reader separately.
64  for (auto&& reader : m_data_readers) {
65  reader->set_comm(m_comm);
66  reader->load();
67  m_data_size += reader->get_linearized_data_size();
68  }
69  // Verify the readers have the same number of samples.
70  int num_samples = m_data_readers[0]->get_num_data();
71  for (auto&& reader : m_data_readers) {
72  if (num_samples != reader->get_num_data()) {
73  throw lbann_exception(
74  "data_reader_merge_features: data readers do not have the same amount of data");
75  }
76  }
77  m_label_reader->load();
78  // Reset indices.
79  m_shuffled_indices.resize(num_samples);
80  std::iota(m_shuffled_indices.begin(), m_shuffled_indices.end(), 0);
81  select_subset_of_data();
82 }

84 bool data_reader_merge_features::fetch_datum(Mat& X, int data_id, int mb_idx,
85  int tid) {
86  int start = 0;
87  for (auto&& reader : m_data_readers) {
88  auto X_view = X(El::IR(start, start + reader->get_linearized_data_size()),
89  El::ALL);
90  reader->fetch_datum(X_view, data_id, mb_idx, tid);
91  start += reader->get_linearized_data_size();
92  }
93  return true;
94 }
